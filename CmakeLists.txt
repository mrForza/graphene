cmake_minimum_required(VERSION 3.14)
project(GrapheneProject)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

# Set compiler flags for MinGW
if(MINGW)
    set(CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} -std=c++17")
    set(CMAKE_C_FLAGS "${CMAKE_C_FLAGS} -std=c11")
endif()

# Include directories
include_directories(${CMAKE_SOURCE_DIR}/algs/general/rb_tree)
include_directories(${CMAKE_SOURCE_DIR}/algs/general/list)

# Source files
set(RB_TREE_SOURCES
    algs/general/rb_tree/rb_tree.cpp
    algs/general/rb_tree/rb_tree.h
)

# Main executable (if needed)
add_executable(main src/main.cpp ${RB_TREE_SOURCES})

# GoogleTest setup
include(FetchContent)
FetchContent_Declare(
    googletest
    GIT_REPOSITORY https://github.com/google/googletest.git
    GIT_TAG v1.14.0
)

# For Windows with Visual Studio: Prevent overriding the parent project's compiler/linker settings
if(MSVC)
    set(gtest_force_shared_crt ON CACHE BOOL "" FORCE)
endif()

FetchContent_MakeAvailable(googletest)

# Test executable
enable_testing()

# Set output directory for test executables
set(TEST_OUTPUT_DIR ${CMAKE_BINARY_DIR}/bin/tests)
file(MAKE_DIRECTORY ${TEST_OUTPUT_DIR})

# Red-Black Tree test
add_executable(rb_tree_test
    algs/general/rb_tree/rb_tree_test.cpp
    ${RB_TREE_SOURCES}
)

target_link_libraries(rb_tree_test
    GTest::gtest
    GTest::gtest_main
)

set_target_properties(rb_tree_test PROPERTIES
    RUNTIME_OUTPUT_DIRECTORY ${TEST_OUTPUT_DIR}
)

# Doubly Linked List sources
set(D_LIST_SOURCES
    algs/general/list/d_list.cpp
    algs/general/list/d_list.h
)

# Doubly Linked List test
add_executable(d_list_test
    algs/general/list/d_list_test.cpp
    ${D_LIST_SOURCES}
)

target_link_libraries(d_list_test
    GTest::gtest
    GTest::gtest_main
)

set_target_properties(d_list_test PROPERTIES
    RUNTIME_OUTPUT_DIRECTORY ${TEST_OUTPUT_DIR}
)

include(GoogleTest)
gtest_discover_tests(rb_tree_test)
gtest_discover_tests(d_list_test)
